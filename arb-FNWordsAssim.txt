! يتم تحويل التاء المربوطة إلى تاء مفتوحة عندما يتبعها أي شيء
ة -> ت || _ "^se^" "+token" ?+ "+token" .o. 

! تتحول الألف اللينة في نحو إلى وعلى ولدى إلى ياء عندما يتبعها أي شيء
ى -> ي || [{إل}|{لد}|{عل}] _ "^se^" "+token" ?+ "+token" .o. 

! تتحول الألف اللينة في عسى وسوى إلى ألف عندما يتبعها أي شيء
ى -> ا || [{عس}|{سو}] _ "^se^ " "+token" ?+ "+token" .o. 

! تتحول الألف واللام في مثل الذي والتي إلى لام فقط عندما يسبقها لام الجر مثل للذي وللتي
{الـ} -> ل ـ ||  ل ـ "^se^" "+token" "^ss^" _ .o.

! يتم حذف إحدى اللامين من اللذان واللذين عندما يسبقها لام الجر مثل للذين وللتين
{لل} -> ل ||  ل ـ "^se^" "+token" "^ss^" _ .o.

! يتم حذف نون عن ومن إذا تلتها نا الفاعلين مثل عن عنا
ن -> 0 ||  _ "^se^" "+token" {ـنا} .o.

! يتم إضافة نون بعد النون المشددة لمثل إن ولكن عند إضافتها إلى ياء المتكلم مثل إنني ولكنني
ن -> ن ن ||  _ "^se^" "^dbl" [2|3] "^dbl" "+token" ـ ي .o.

! يتم حذف التنوين من الأف اختيارا مثل أيضاَ أيضا
ً (->) 0  .o.

! يتم حذف نون عن ومن إذا تلتها من وما الموصولتين مثال مما وعمن
ن -> 0 ||  _ "^se^" "+token" "^ss^" م .o.

! الياء تحذف إذا تبعها ياء المتكلم مثل إلى إلي وفي في
ي -> 0 ||  _ "^se^" "+token" ـ ي
